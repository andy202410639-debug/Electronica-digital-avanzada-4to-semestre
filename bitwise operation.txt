
//Codigo para identificar todas las potencias del 2 del (0 al 100) utilizando bitwise operation.

#include <stdio.h> 

int main() {
    
    int i; //Se declara la variable

    printf("Potencias del 2 entre 0 al  100:\n");  //Se indica que va hacer el codigo 

    for (i = 0; i <= 100; i++) {  // Se utiliza un ciclo que recorre todos los números desde 0 hasta 100 evaluando cada uno.
        
        if (i > 0 && (i & (i - 1)) == 0) { //Se aplica una operación bitwise AND - (i & (i - 1))
            
            printf("%d\n", i); //Se imprime el resultado 
        }
    }

    return 0; //Finaliza mi codigo 
}
